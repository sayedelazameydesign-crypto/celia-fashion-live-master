    cd EcommerceAI3D
    python -m venv venv
    venv\Scripts\activate  (Windows)
    source venv/bin/activate  (Mac/Linux)
    uv pip install -r requirements.txt

2️⃣  إعداد البيئة:
    cp .env.example .env
    # عدل .env بمعلومات قاعدة البيانات

3️⃣  تشغيل المشروع:
    python run.py
    
    ثم افتح: http://localhost:5000
